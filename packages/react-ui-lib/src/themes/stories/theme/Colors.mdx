import { Meta, Canvas, ColorPalette, ColorItem, Story } from '@storybook/blocks';
import { useTheme, ThemeProvider } from "@mui/material";

import themes from "../../Standard"

<Meta title="theme/Colors"/>

export function MainColors({isDark = false }) {
    const theme = isDark ? themes.dark:themes.light;
    return (
        <ThemeProvider theme={theme}>
          <ColorPalette>
              <ColorItem
                  title="theme.pallete.primary"
                  subtitle="Primary"
                  colors={{
                      light: theme?.palette?.primary?.light,
                      main: theme?.palette?.primary?.main,
                      dark: theme?.palette?.primary?.dark,
                      contrast: theme?.palette?.success?.contrastText,
                  }}
              />
              <ColorItem
                  title="theme.pallete.secondary"
                  subtitle="Secondary"
                  colors={{
                      light: theme?.palette?.secondary?.light,
                      main: theme?.palette?.secondary?.main,
                      dark: theme?.palette?.secondary?.dark,
                      contrast: theme?.palette?.success?.contrastText,
                  }}
              />
          </ColorPalette>
        </ThemeProvider>
    );
}

export function BackgroundColors({isDark = false }) {
    const theme = isDark ? themes.dark:themes.light;
